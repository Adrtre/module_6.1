SELECT i.ID, i.SUMMARY, i.DESCRIPTION, il.NAME AS ISSUESLIST_NAME
FROM ISSUES i
         JOIN ISSUESLISTS il ON i.ISSUESLIST_ID = il.ID;

SELECT i.ID, i.SUMMARY, i.DESCRIPTION, u.FIRSTNAME , u.LASTNAME
FROM ISSUES i
         LEFT JOIN USERS u ON i.USER_ID_ASSIGNEDTO = u.ID;

SELECT USERS.FIRSTNAME,USERS.LASTNAME, COUNT(ISSUES.ID) AS TASK_COUNT
FROM USERS
         LEFT JOIN ISSUES ON USERS.ID = ISSUES.USER_ID_ASSIGNEDTO
GROUP BY USERS.LASTNAME, USERS.FIRSTNAME;
